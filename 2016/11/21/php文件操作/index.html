删除文件夹

// $res = rmdir('test1');  //这个只能删除空文件夹


//删除文件夹及内部文件
function removeDir($path){
    $arr = scandir($path);
//    print_r($arr);die;
    if(count($arr)>2){
        //目录不为空
        for($i=2;$i<count($arr);$i++){
             $url = $path.'/'.$arr[$i];
            //$arr[$i] = 1.txt  test1
             if(is_dir($url)){
                 //为目录. 
                 removeDir($url);
             }else{
                 //是文件
                unlink($url);
             }
        }
    }
    $res = rmdir($path);
    return $res;
}
